遺伝子型と表現型の関係を推定することは重要なタスクとなっています。一方でGWAS解析などの結果、遺伝子変異だけでは全ての表現型の変化を捉えられないことがわかってきています。様々な実験結果から、非翻訳領域が遺伝子発現に影響を与えていることがわかっています。遺伝子発現は細胞の種類や状態を決定するものであり、遺伝子発現は扱うことが可能な中間の表現型とみなすことができます。 配列を用いたエピジェネティクス解析、転写プロファイル解析のディープラーニングを使った例を紹介します。配列から、DNASeqやChipSeqなどを用いて発現領域を求めることが可能となっています。今回は配列から、どこが発現領域かを推定するタスクを考えてみます。

class RootBlock(chainer.Chain):

  def __init__(self, out_ch, kerne, pool)
    super(RootBlock, self).__init__()

    self.pool = pool
    with self.init_scope():
        pad = kernel // 2
        self.conv = L.ConvolutionND(1, None, out_ch, kernel, pad=pad, nobias=True)
        self.bn = L.BatchNormalization(out_ch)

  def forward(self, x):
    h = self.conv(x)
    h = self.bn(h)
    h = F.max_pooling_nd(h, self.pool, self.pool, 0, cover_all=True)
    return F.relu(h)

class DilatedBlock(chainer.Chain):
  def __init__(self, out_ch, kernel, dilate):
    super(DilatedBlock, self).__init__()
    with self.init_scope():
      self.conv = L.ConvolutionND(1, None, out_ch, kernel, pad=dilate, nobias=True, dilate=dilate)
      self.bn = L.BatchNormalization(out_ch)

  def forward(self, x):
    h = self.conv(x)
    h = self.bn(h)
    return F.relu(h)

import chainer.functions as F
import math
import sklearn

def log_poisson_loss(log_x, t):
  return F.mean(F.exp(log_x) - t * log_x)


def log_poisson_acc(log_x, t):
  xp = cp.get_array_module(t)
  t = t.astype(xp.float32)
  zeros = xp.zeros_like(t, dtype=t.dtype)
  ones = xp.ones_like(t, dtype=t.dtype)
  cond = xp.logical_and(t >= zeros, t <= ones)
  stirling_approx = t * F.log(t) - t + 0.5 * F.log(2.0 * math.pi * t)
  ret = F.exp(log_x) - t * log_x + F.where(cond, zeros, stirling_approx)
  return F.mean(ret)
